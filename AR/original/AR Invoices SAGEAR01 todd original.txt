SELECT 
  T.INTERFACE_STATUS_F CNTBTCH,
  ROW_NUMBER() OVER(PARTITION BY T.INTERFACE_STATUS_F ORDER BY T.INTERFACE_STATUS_F, C.USER3, T.BILL_NUMBER) CNTITEM,
  C.USER3 IDCUST,
  T.BILL_NUMBER IDINVC,
  '1' TEXTTRX,
  (SELECT TRACE_NUMBER FROM TRACE WHERE TRACE_TYPE = 'P' AND DETAIL_NUMBER = T.DETAIL_LINE_ID LIMIT 1) CUSTPO,
  T.TOTAL_CHARGES AMTINVCTOT,
  '1' SWMANTX
FROM TLORDER T
JOIN CLIENT C ON C.CLIENT_ID = T.BILL_TO_CODE
WHERE 1 = 1
  AND T.INTERFACE_STATUS_F > 0
  AND T.EXTRA_STOPS <> 'Child'
  AND T.INTERFACE_STATUS_F = :AUDIT
ORDER BY 1, 3, 4
